========================================================================
NEXT STEPS - To complete the setup:
========================================================================

  1. Create a secret with your ADCS credentials (if not using adcs.credentials.create):

     kubectl create secret generic adcs-credentials \
       --from-literal=username='DOMAIN\\username' \
       --from-literal=password='your-password' \
       -n {{ .Release.Namespace }}

  2. Create an AdcsIssuer or ClusterAdcsIssuer:

     cat <<EOF | kubectl apply -f -
     apiVersion: adcs.certmanager.lcwsre.io/v1
     kind: ClusterAdcsIssuer
     metadata:
       name: my-adcs-issuer
     spec:
       url: https://your-adcs-server.example.com/certsrv
       credentialsRef:
         name: adcs-credentials
       caBundle: <BASE64_ENCODED_CA_CERT>
     EOF

  Or use the all-in-one approach with helm upgrade:

     helm upgrade {{ .Release.Name }} adcs-issuer \
       --set adcs.credentials.create=true \
       --set adcs.credentials.username='DOMAIN\\user' \
       --set adcs.credentials.password='password' \
       --set adcs.issuer.create=true \
       --set adcs.issuer.kind=ClusterAdcsIssuer \
       --set adcs.issuer.url='https://adcs.example.com/certsrv' \
       --set-file adcs.issuer.caBundle=ca-bundle.pem
{{- end }}
